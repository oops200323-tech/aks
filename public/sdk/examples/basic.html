Advanced configuration with custom colors and callbacks:

new NPSWidget({ apiKey: 'nps_your_api_key_here', // Positioning & Styling position: 'bottom-right', primaryColor: '#FF6B6B', roundedCorners: true, closeButton: true, progressBar: true, // Timing delay: 3000, autoFadeOut: true, fadeOutDelay: 5000, // Behavior showOnce: false, autoRotate: true, // URL Targeting urlTargeting: { enabled: true, mode: 'include', patterns: [ { type: 'contains', value: '/product' }, { type: 'starts_with', value: '/pricing' } ] }, // SPA Support spaMode: true, // Event Callbacks onResponse: function(data) { console.log('Score:', data.score); // Send to analytics gtag('event', 'nps_response', { value: data.score }); }, onShow: function(survey) { console.log('Widget shown:', survey.name); }, onClose: function() { console.log('Widget closed'); }, onError: function(message, error) { console.error('Error:', message, error); } });
